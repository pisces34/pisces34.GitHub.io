``` C++
#include<cstdio>
#include<iostream>
using namespace std;
#define Max 100001
int main(){
    int N;
    int A[Max],tempIndex=0;
    scanf("%d",&N);
    int ThisSum, MaxSum,i;
    ThisSum = 0;
    MaxSum = -1;
    int left=0,right=N-1;
    for (i = 0; i < N; i++ ) {
        scanf("%d", &A[i]);
        ThisSum += A[i];
        if(ThisSum < 0) {
            ThisSum = 0;    //变小则清0
            tempIndex = i+1; //移动边界
        }else if(ThisSum > MaxSum) {
            MaxSum = ThisSum;
            left = tempIndex; 
            right = i;
        }
    }
    if (MaxSum<0) {
        MaxSum = 0;
    }
    printf("%d %d %d",MaxSum, A[left], A[right]);
    return 0;
}
```
