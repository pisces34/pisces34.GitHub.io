```C++
#include <iostream>
#include "cstdio"
#include "map"
using namespace std;

int main() {
    int k;
    map<int, float, greater<int>> nums;
    cin>>k;
    int exp, count = 0;
    float coef;
    for (int i = 0; i < k; ++i) {
        cin>>exp>>coef;
        nums.insert({exp,coef});
    }
    cin>>k;
    for (int i = 0; i < k; ++i) {
        cin>>exp>>coef;
        if (nums.find(exp) != nums.end()) {
            nums.find(exp)->second += coef;
        }else{
            nums.insert({exp,coef});
        }
    }
    for(auto &t: nums){
        if (t.second != 0){
            count++;
        }

    }
    cout<<count;
    for(auto &t: nums){
        if (t.second != 0) {
            printf(" %d %.1lf", t.first, t.second);
        }
    }
    return 0;
}
```
