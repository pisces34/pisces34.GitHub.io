```C++
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    int a,b, sum = 0;
    cin>>a>>b;
    sum = a+b;
    string s = to_string(sum);
    if (s.size()<4){
        cout<<sum;
    }else {
        string res = "";
        int count = 0;
        for (int i = s.size() - 1; i > 0; --i) {
            res += s[i];
            count++;
            if (count == 3) {
                if (s[i-1] != '-' || (s[i-1]>='0')) res += ',';
                count = 0;
            }
        }
        res += s[0];
        std::reverse(res.begin(), res.end());
        cout << res;
    }
    return 0;
}
```
