class Solution {
    public int sumOfUnique(int[] nums) {
        int[] array = new int[101];
        for (int i : nums) {
            array[i]++;
        }
        int ans = 0;
        for (int i = 0; i < 101; i++) {
            if (array[i] == 1) {
                ans += i;
            }
        }
        return ans;
    }
}
