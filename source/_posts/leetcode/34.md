``` java
class Solution {
    public int[] searchRange(int[] nums, int target) {
                    int[] res = new int[]{-1,-1};
            int left= 0, right = nums.length;
            if (right > 0) {
                for (int i = 0; i < right; i++) {
                    if (nums[i] == target) {
                        left = i;
                        res[0] = left;
                        break;
                    }
                }
                for (int i = right-1; i >= left; i--) {
                    if (nums[i] == target) {
                        res[1] = i;
                        break;
                    }
                }
            }
            return res;
    }
}
```
