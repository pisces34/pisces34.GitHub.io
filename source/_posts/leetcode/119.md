首先想到的是用二维数组保存记录

``` swift
class Solution {
    func getRow(_ rowIndex: Int) -> [Int] {
        if rowIndex == 0 {
            return [1]
        }
        var dp = [[Int]]()
        var row = [Int]()
        row.append(1)
        dp.append(row)
        row = []
        for i in 1 ... rowIndex {
            row.append(1)
            let n = dp[i-1].count - 1
            for j in 0 ..< n {
                row.append(dp[i-1][j]+dp[i-1][j+1])
            }
            row.append(1)
            dp.append(row)
            row = []
        }
        return dp[rowIndex]
    }
}
```

O(rowIndex) 空间复杂度， 用两个一维数组

``` swift
class Solution {
    func getRow(_ rowIndex: Int) -> [Int] {
        if rowIndex == 0 {
            return [1]
        }
        var res = [Int]()
        var row = [Int]()
        for i in 1 ... rowIndex {
            for j in 0 ... i {
                if j == 0 || j == i {
                    row.append(1)
                }else{
                    row.append(res[j-1] + res[j])
                }
            }
            res = row
            row = []
        }
        return res
    }
}
```
