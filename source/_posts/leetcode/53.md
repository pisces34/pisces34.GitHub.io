贪心算法
``` C++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int sum = INT_MIN;
        int thisSum = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (thisSum < 0){
                thisSum = 0;
            }
            thisSum += nums[i];
            sum = max(sum,thisSum);
        }
        return sum;
    }
};
```

动态规划
``` C++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        //res的初值也可设置为最小值
        //nums[0]可能是最小值且为负数
        int pre = 0, res = nums[0]; 
        for (auto num: nums) {
            pre = max(pre + num, num);
            res = max(pre, res);
        }
        return res;
    }
};
```
