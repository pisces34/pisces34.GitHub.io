### Java
``` Java
class Solution {
    public int lengthOfLastWord(String s) {
        int res = 0;
        String ss = s.trim();
        int n = ss.length();
        while(n >= 1) {
            if (ss.charAt(n-1) != ' ') {
                res++;
                n--;
            } else {
                break;
            }
        }
        return res;
    }
}
```
