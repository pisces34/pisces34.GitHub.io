class Solution {
    func findLengthOfLCIS(_ nums: [Int]) -> Int {
        var arr = nums.sorted()
        var count = 1
        var res = 1
        for i in 0 ..< nums.count - 1 {
            if nums[i] < nums[i + 1] {
                count += 1
            } else {
                count = 1
            }
            res = max(res, count)
        }
        return res
    }
}
