### 代码
``` swift
class Solution {
    func isGood(_ nums: [Int]) -> Bool {
        let n = nums.count;
        var sortednums = nums.sorted()
        for i in 0 ..< n - 1 {
            if sortednums[i] != i + 1 {
                return false;
            }
        }
        if sortednums[n - 1] == n - 1 {
            return true;
        }
        return false;
    }
}
```
