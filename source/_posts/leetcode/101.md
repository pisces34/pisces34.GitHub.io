class Solution {
    func isSymmetric(_ root: TreeNode?) -> Bool {

        func check(_ l: TreeNode?, _ r: TreeNode?) -> Bool {
            if l == nil && r == nil {
                return true
            }
            
            if l == nil || r == nil {
                return false
            }
            
            if l?.val == r?.val && check(l?.left, r?.right) && check(l?.right, r?.left) {
                return true
            }
            return false
        }
        return check(root, root)
    }
}
