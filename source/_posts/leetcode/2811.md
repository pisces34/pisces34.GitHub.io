### JavaScript
``` JavaScript
/**
 * @param {number[]} nums
 * @param {number} m
 * @return {boolean}
 */
var canSplitArray = function (nums, m) {
    const length = nums.length;
    if (length <= 2) {
        return true;
    }
    for (let i = 0; i < length - 1; i++) {
        if (nums[i] + nums[i + 1] >= m) {
            return true;
        }
    }

    return false;
};
```
