``` java
    class Solution {
        public List<String> simplifiedFractions(int n) {
            List<String> list = new LinkedList<>();
            if (n == 1) {
                return list;
            }
            for (int i = 1; i < n; i++) {
                for (int j = i+1; j <= n; j++) {
                    if (gcd(i,j) == 1) {
                        list.add(i+"/"+j);
                    }
                }
            }
            return list;
        }
        public int gcd(int a, int b) {
            int c = a % b;
            while (c != 0) {
                a = b;
                b = c;
                c = a % b;
            }
            return b;
        }
    }
    ```
