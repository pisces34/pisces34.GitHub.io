class Solution {
    func areNumbersAscending(_ s: String) -> Bool {
        let array = s.split(separator: " ")
        var last = Int.min
        for item in array {
            if let myInt = Int(item) {
                if myInt > last {
                    last = myInt
                }else{
                    return false
                }
            }
        }
        return true
    }
}
