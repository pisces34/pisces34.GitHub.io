class Solution {
    func getMaximumGenerated(_ n: Int) -> Int {
        var nums = [Int].init(repeating: 0, count: n+2)
        var odd = 1, flag = true
        if n == 0 {
            return 0
        }else if n == 1 {
            return 1
        }
        nums[0] = 0
        nums[1] = 1
        for i in 2 ... n {
            if flag {
                nums[i] = nums[i/2]
                flag = false
            } else {
                nums[i] = nums[odd] + nums[odd+1]
                odd += 1
                flag = true
            }
        }
        nums.sort(by: {$0 > $1})
        return nums[0]
    }
}
