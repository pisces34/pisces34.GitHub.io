class Solution {
    func balancedStringSplit(_ s: String) -> Int {
        var arr = [Character](s)
        var r = 0, res = 0
        let n = s.count
        for i in 0 ..< n {
            if arr[i] == "R" {
                r += 1
            } else {
                r -= 1
            }
            if r == 0 {
                res += 1
            }
        }
        return res
    }
}
