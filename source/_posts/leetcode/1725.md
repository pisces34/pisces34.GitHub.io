class Solution {
    func countGoodRectangles(_ rectangles: [[Int]]) -> Int {
        var maxLen = 0
        var count = 0
        for nums in rectangles {
            let l = nums[0], r = nums[1]
            let k = min(l,r)
            if k == maxLen {
                count += 1
            } else if k > maxLen {
                maxLen = k
                count = 1
            }
        }
        return count
    }
}
