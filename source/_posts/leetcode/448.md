``` TypeScript
function findDisappearedNumbers(nums: number[]): number[] {
    let n = nums.length;
    nums.forEach(num => {
        let x = (num - 1) % n;
        nums[x] += n;
    })
    let res: number[] = [];
    for (let i = 0; i < n; i++) {
        if (nums[i] <= n) {
            res.push(i + 1);
        }
    }
    return res;
};
```
