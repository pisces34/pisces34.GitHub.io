class MyHashSet {
        var HashTable: [Int]
        let p = 1000003
        init() {
            HashTable = [Int].init(repeating: 1000003, count: 1000003)
        }

        func add(_ key: Int) {
            if HashTable[key] == p {
                HashTable[key] = key
            }
        }

        func remove(_ key: Int) {
            if HashTable[key] == key {
                HashTable[key] = p
            }
        }

        func contains(_ key: Int) -> Bool {
            if HashTable[key] == key {
                return true
            }
            return false
    }
}
