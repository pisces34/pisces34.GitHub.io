class Solution {
    func numberOfBoomerangs(_ points: [[Int]]) -> Int {
        var ans = 0
        let n = points.count
        for i in 0 ..< n {
            var hash: [Int: Int]=[:]
            for j in 0 ..< n {
                let x = points[i][0] - points[j][0]
                let y = points[i][1] - points[j][1]
                let dist = x*x + y*y
                if hash[dist] != nil {
                    hash[dist]! += 1
                }else {
                    hash[dist] = 1
                }
            }
            for m in hash.values {
                ans = ans + m*(m-1)
            }
        }
        return ans
    }
}
