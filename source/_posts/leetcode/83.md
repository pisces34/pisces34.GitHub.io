``` swift
class Solution {
    func deleteDuplicates(_ head: ListNode?) -> ListNode? {
        if head == nil || head?.next == nil{
            return head
        }
        var cur = head
        while cur?.next != nil {
            //当前值与下一个节点值重复
            if cur?.val == cur?.next?.val {
                //使得当前节点指向下下一个节点
                cur?.next = cur?.next?.next
            }else {
                //不相等则移动当前指针
                cur = cur?.next
            }
        }
        return head
    }
}
```
