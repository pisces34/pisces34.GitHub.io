### 代码

```swift
class MinStack {
    var stack = [Int]()
    var point: Int
    var minNum: Int
    init() {
        stack = []
        point = -1
        minNum = Int.max
    }

    func push(_ val: Int) {
        point += 1
        stack.append(val)
        if val <= minNum {
            minNum = val
        }
    }

    func pop() {
        if point >= 0 {
            let num = stack.removeLast()
            point -= 1
            if num == minNum {
                minNum = Int.max
                for i in stride(from: point, through: 0, by: -1) {
                    minNum = min(stack[i], minNum)
                }
            }
        }
    }

    func top() -> Int {
        if point > -1 {
            return stack[point]
        }
        return -1
    }

    func getMin() -> Int {
        if point > -1 {
            return minNum
        }
        return -1
    }
}

```

class MinStack {
    var data = [Int]()
    var help = [Int]()
    init() {
        data = []
        help = []
    }

    func push(_ val: Int) {
        data.append(val)
        var tail = 0
        if  help.last != nil {
            tail = help.last!
        }
        if help.isEmpty || tail >= val {
            help.append(val)
        } else {
            help.append(tail)
        }
        
    }

    func pop() {
        data.removeLast()
        help.removeLast()
    }

    func top() -> Int {
        if let lastone = data.last {
            return lastone
        }
        return -1
    }

    func getMin() -> Int {
        if let minNum = help.last {
            return minNum
        }
        return -1
    }

}
