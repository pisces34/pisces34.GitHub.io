给定一个大小为 n 的数组，找到其中的多数元素。多数元素是指在数组中出现次数 大于 ⌊ n/2 ⌋ 的元素。
你可以假设数组是非空的，并且给定的数组总是存在多数元素。

### 示例 1：
- 输入：[3,2,3]
- 输出：3

### 示例 2：
- 输入：[2,2,1,1,1,2,2]
- 输出：2

``` swift
class Solution {
    func majorityElement(_ nums: [Int]) -> Int {
        if nums.count == 1 {
            return nums[0]
        }
        let size = nums.count/2 + 1
        var arr = [Int : Int]()
        var res = 0
        for i in nums {
            if let value = arr[i] {
                arr[i]! += 1
                if arr[i]! >= size {
                    res = i
                }
            }else{
                arr[i] = 1
            }
        }
        return res
    }
}
```

``` java
class Solution {
    public int majorityElement(int[] nums) {
        int count = 0, compare = 0;
        for(int i : nums) {
            if (count == 0) {
                compare = i;
            }
            // 判断当前元素是否和比较中的元素相等，不相等则次数-1
            count += i == compare ? 1 : -1;
        }
        return compare;
    }
}
```
