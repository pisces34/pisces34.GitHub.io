``` java
class Solution {
    public int findMinFibonacciNumbers(int k) {
        ArrayList<Integer> array = new ArrayList<>();
        int last = 0, cur = 1, next = 0;
        while (cur <= k) {
            array.add(cur);
            next = last + cur;
            last = cur;
            cur = next;
        }
        int count = 0;
        for (int i = array.size()-1; i >= 0 ; i--) {
            if (k - array.get(i) >= 0) {
                k -= array.get(i);
                count++;
            }
        }
        return count;
    }
}
```
