### 递归
``` swift
class Solution {
    func insertIntoBST(_ root: TreeNode?, _ val: Int) -> TreeNode? {
        if root == nil {
            return TreeNode.init(val)
        }
        if val > root!.val {
            root?.right = insertIntoBST(root?.right, val)
        }
        if val < root!.val {
            root?.left = insertIntoBST(root?.left, val)
        }
        return root
    }
}
```
