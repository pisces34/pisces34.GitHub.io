class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int res = 0;
        int maxV = nums[0], minV = nums[0];
        int left = 0,right = 0;
        int n = nums.length;
        while (left <= right && right < n){
            if (nums[right] > maxV) {
                maxV = nums[right];
            }
            if (nums[right] < minV) {
                minV = nums[right];
            }
            if (maxV - minV <= limit) {
                res = Math.max(res, right-left+1);
                right++;
            } else {
                left++;
                minV = 1000000001;
                maxV = 0;
                for (int i = left; i <= right ; i++) {
                    if (nums[i] > maxV) {
                        maxV = nums[i];
                    }
                    if (nums[i] < minV) {
                        minV = nums[i];
                    }
                }
                res = Math.max(res, right-left+1);
                right++;
            }
        }
        return res;
    }
}
