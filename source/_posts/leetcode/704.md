class Solution {
    func search(_ nums: [Int], _ target: Int) -> Int {
        var left = 0, right = nums.count - 1
        var index = -1
        while left <= right {
            index = left + (right-left)/2
            if nums[index] == target {
                return index
            }
            if nums[index] < target {
                left = index + 1
            }else {
                right = index - 1
            }
        }
        return -1
    }
}
