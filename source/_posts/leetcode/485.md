class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int res = 0;
        int count = 0;
        for (int num:nums) {
            if (num == 1) {
                count ++;
            } else {
                res = Math.max(res, count);
                count = 0;
            }
        }
        return res > count ? res : count;
    }
}
