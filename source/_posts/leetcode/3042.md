function countPrefixSuffixPairs(words: string[]): number {
    let res = 0;
    for (let i = 0; i < words.length - 1; i++) {
        for (let j = i + 1; j < words.length; j++) {
            if (isPrefixAndSuffix(words[i], words[j])) {
                res ++;
            }
        }
    }
    return res;
};
function isPrefixAndSuffix(str1, str2): boolean {
    let i = 0, j = 0;
    while(i < str1.length) {
        if (str1[i] != str2[j]) {
            return false;
        }
        i++;
        j++;
    }
    i = 0, j = str2.length - str1.length;
    while(i < str1.length) {
        if (str1[i] != str2[j]) {
            return false;
        }
        i++;
        j++;
    }
    return true;
}
