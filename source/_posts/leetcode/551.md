class Solution {
    func checkRecord(_ s: String) -> Bool {
        var A = 0, L = 0
        for i in s {
            if i == "A" {
                A += 1
                L = 0
            } else if i == "L" {
                L += 1
            } else {
                L = 0
            }
            if A > 1 || L > 2 {
                return false
            }
        }
        return true
    }
}
