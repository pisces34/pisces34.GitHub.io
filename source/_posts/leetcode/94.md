class Solution {
    var res: [Int] = []
    func inorderTraversal(_ root: TreeNode?) -> [Int] {
        guard let node = root else {
            return []
        }
        inorderTraversal(node.left)
        res.append(node.val)
        inorderTraversal(node.right)
        return res
    }
}
