class Solution {
    func merge(_ nums1: inout [Int], _ m: Int, _ nums2: [Int], _ n: Int) {
        var i = m + n - 1
        var long = m - 1, short = n - 1
        while short >= 0 {
            while long >= 0 && nums1[long] > nums2[short] {
                nums1[i] = nums1[long]
                i -= 1
                long -= 1
            }
            nums1[i] = nums2[short]
            i -= 1
            short -= 1
        }
    }
}
