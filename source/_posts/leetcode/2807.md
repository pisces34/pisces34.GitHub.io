### JavaScript
``` JavaScript
var insertGreatestCommonDivisors = function(head) {
    let cur = head;
    while(cur.next != null) {
        let insertNode = new ListNode(gcd(cur.val, cur.next.val));
        insertNode.next = cur.next;
        cur.next = insertNode;
        cur = insertNode.next;
    }
    return head;
};
function gcd(a,b){
  if(b === 0) return a;
  return gcd(b,a % b);
}
```
