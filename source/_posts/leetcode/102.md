class Solution {
    func levelOrder(_ root: TreeNode?) -> [[Int]] {
        if root == nil {
            return []
        }
        var res = [[Int]]()
        var queue = [TreeNode]()
        queue.append(root!)
        while !queue.isEmpty {
            var row = [Int]()
            for i in 0 ..< queue.count {
                let T = queue.removeFirst()
                row.append(T.val)
                if let node = T.left {
                    queue.append(node)
                }
                if let node = T.right {
                    queue.append(node)
                }
            }
            res.append(row)
        }
        return res
    }
}
