class Solution {
    public int countKDifference(int[] nums, int k) {
        int[] cnts = new int[110];
        int n = nums.length, ans = 0;
        for (int i = 0; i < n; i++) {
            int t = nums[i];
            // 两数之差在cnts中存在则ans+1
            if (t - k >= 1) {
                ans += cnts[t - k];
            }
            // 两数之和在cnts中存在则ans+1
            if (t + k <= 100) {
                ans += cnts[t + k];
            }
            cnts[t]++;
        }
        return ans;
    }
}
