```swift
class Solution {
    func chalkReplacer(_ chalk: [Int], _ k: Int) -> Int {
        var who = 0, rest = k, n = chalk.count
        if n == 1 {
            return 0
        }
        var sum = chalk.reduce(0, +)
        var k = k % sum
        for i in 0 ..< n {
            if chalk[i] > k {
                who = i
                break
            }
            k -= chalk[i]
        }
        return who
    }
}
```
