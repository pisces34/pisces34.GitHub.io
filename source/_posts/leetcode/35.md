class Solution {
    func searchInsert(_ nums: [Int], _ target: Int) -> Int {
        var left = 0, right = nums.count - 1
        while left <= right {
            var mid = left + (right - left)/2
            if nums[mid] < target {
                left = mid + 1
            }else{
                right = mid - 1
            }
        }
        return left
    }
}
