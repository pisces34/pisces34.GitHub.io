 class Solution {
    func merge(_ intervals: [[Int]]) -> [[Int]] {
        if intervals.count < 2 { return intervals }
        var l = 0, r = 0, j = 0
        let sortedArr = intervals.sorted { $0[0] < $1[0] }
        var res = [[Int]]()
        res.append(sortedArr[0])
        for item in sortedArr {
            if res[j][1] < item[0] {
                res.append(item)
                j += 1
            }else{
                res[j][1] = max(res[j][1], item[1])
            }
        }
        return res
    }
}
