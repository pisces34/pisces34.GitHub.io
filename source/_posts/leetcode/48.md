class Solution {
    func rotate(_ matrix: inout [[Int]]) {
        let n = matrix.count
        for i in 0 ..< n {
            matrix[i].reverse()
        }
        var left =  1, bottom = 0
        for i in stride(from: n-1, to: 0, by: -1) {
            //向上一层
            bottom = i - 1
            for j in left ..< n {
                let tmp = matrix[i][j]
                matrix[i][j] = matrix[bottom][left-1]
                matrix[bottom][left-1] = tmp
                bottom -= 1
            }
            //向右一列
            left += 1
        }
    }
}
