class Solution {
    func longestPalindrome(_ s: String) -> Int {
        var dict = [Character : Int]()
        var count = 0
        for char in s {
            if dict[char] != nil {
                dict[char]! += 1
            } else {
                dict[char] = 1
            }
        }
        for v in dict.values {
            count += v / 2 * 2
            //当出现一次奇数个字符值时+1作为回文中心
            if v % 2 == 1 && count % 2 == 0 {
                count += 1
            }
        }
        return count
    }
}
