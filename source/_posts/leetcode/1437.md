class Solution {
    public boolean kLengthApart(int[] nums, int k) {
        // 设置初始间距
        int distance = k;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 1) {
                if (distance < k) {
                    return false;
                } else {
                    distance = 0;
                }
            } else {
                distance++;
            }
        }
        return true;
    }
}
