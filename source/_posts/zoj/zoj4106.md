``` C++
#include <iostream>
using namespace std;
int main() {
    int  N;
    int m,n;
    cin>>N;
    while (N--){
        cin>>n;
        if (n<=7){
            m=7;
        }else if(n%7==0&&n%4!=0){
            m = n;
        }else{
            m = (n/7)*7+7;
            if (m%4==0){
                m = m+7;
            }
        }
        cout<<m<<endl;
    }
    return 0;
}
```

``` C++
#include <iostream>
using namespace std;
int main() {
    int  N;
    int m,n;
    cin>>N;
    while (N--){
        cin>>n;
        for (int i = n; i <= 105; ++i) {
            if (i%7==0 && i%4!=0){
                cout<<i<<endl;
                break;
            }
        }
    }
    return 0;
}
```
