``` C++
#include <iostream>
using namespace std;
int main() {
    int T;
    string arr = "0aeiyou";
    cin>>T;
    string str;
    while (T--){
        cin>>str;
        string res="";
        res+=str[0];
        for (int i = 1; i < str.length(); ++i) {
            if(arr.find(str[i]) == string::npos){
                res+=str[i];
            }
        }
        cout<<res<<endl;
    }
    return 0;
}
```
